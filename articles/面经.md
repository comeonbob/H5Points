      if (!Array.isArray(arr) || arr.length === 0 ) {

      let last = [...arr[0]]
      for (let i = 1, n = arr.length; i < n; i++) {

        last = temp
      }

    function dicarle2(arr) {
      if (!Array.isArray(arr) || arr.length === 0 ) {

        let temp = []
        for (let item1 of arr1) {

            temp.push(item1 + '' + item2)
          }
        }
        return temp
      }
    }

    function dicarle3(arr, low, high) {
